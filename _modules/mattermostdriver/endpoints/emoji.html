[docs]	def create_custom_emoji(self, emoji_name, files):
		emoji = {'name': emoji_name, 'creator_id': self.client.userid}
		return self.client.post(
			self.endpoint,
			data={'emoji': json.dumps(emoji)},
			files=files
		)
